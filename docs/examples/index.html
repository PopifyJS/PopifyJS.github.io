Example #1

Case: using the popup in a delete situation... (on delete button click, the popup activates)

<button class="delete-btn">Delete Something</button>
const deleteBtn = document.querySelector(".delete-btn");

deleteBtn.addEventListener("click", () => {
  
  popify({
    headerContent: "Confirm Delete",
    subText: "are you sure you want to delete this thing?",
    overlay: true,
    short: true,
    closeOnEscape: true,
    icon: "error",
    closeIcon: true,
    buttons: [
      {
        text: "Cancel",
        icon: "x lg",
        closePopup: true,
        run: () => {
          alert("Cancelled delete!");
        }
      },
      {
        text: "Delete",
        icon: "trash",
        type: "error",
        closePopup: true,
        run: () => {
          // do the delete function to delete the thing.
          alert("Deleted!");
        }
      }
    ]
  });

});
